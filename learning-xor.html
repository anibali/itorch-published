Learning XOR¶

Our aim is to train a model to learn the following function:

$$ output = xor(input_1, input_2) = \begin{cases} -1 & input_1 \times input_2 \gt 0 \\ 1 & otherwise \end{cases} $$

function visualise_model(model)
    local filename = os.tmpname()
    graph.dot(model.fg, 'Title', filename)
    os.remove(filename .. '.dot')

    local file = io.open(filename .. '.svg', 'r')
    local image_b64 = base64.encode(file:read('*all'))
    file:close()
    os.remove(filename .. '.svg')

    itorch.html('<img src=data:image/svg+xml;base64,' .. image_b64 .. '>')
end

Training data¶

Define function to learn¶

In [3]:
function xor(input_1, input_2)
  if input_1 * input_2 > 0 then
    return -1
  else
    return 1
  end
end

Create training data¶

In [4]:
-- Number of training examples
local n = 800

training_data = {}

-- Create list of n inputs as random pairs of numbers
training_data.inputs = _.map(_.range(1, n), function()
  return torch.randn(2)
end)

-- Create list of outputs as XOR of input values
training_data.outputs = _.map(training_data.inputs, function(i, input)
    return torch.Tensor{xor(input[1], input[2])}
end)

-- Add size method to dataset
function training_data:size()
    return n
end

-- Get the model parameters and their gradients
-- x = flattened model parameters
local x, dloss_dx = model:getParameters()

-- Keep track of current training example index between
-- calls to the evaluation function
local index = 0

-- Define evaluation function as a function that closes
-- over the current sample index, model parameters and
-- their gradients, the model, the criterion, and the
-- training dataset
local feval = function(x_new)
    if x ~= x_new then
        x:copy(x_new)
    end
    
    -- Pull a sample from the training dataset
    index = index + 1
    if index > training_data:size() then index = 1 end
    local input = training_data.inputs[index]
    local target_output = training_data.outputs[index]
    
    -- Zero the gradient
    dloss_dx:zero()
    
    -- Forward pass to calculate loss
    local model_output = model:forward(input)
    local loss = criterion:forward(model_output, target_output)
    
    -- Backward pass to modify weights (backpropagation)
    local dloss_dout = criterion:backward(model_output, target_output)
    model:backward(input, dloss_dout)

    return loss, dloss_dx
end

-- Here we will use stochastic gradient descent (SGD)

-- Initialise optimiser state with SGD parameters
local state = {
    learningRate = 1e-2,
    learningRateDecay = 1e-5,
    momentum = 0.8
}

-- Define optimisation step as a function that closes over
-- the evaluation function, model parameters, and optimiser
-- state
function do_optimisation_step()
    return optim.sgd(feval, x, state)
end

-- The number of times to loop over the training dataset
local n_epochs = 200

local function do_training_epoch(i)
    -- This variable is used to estimate the average loss
    local current_loss = 0

    -- Each epoch is a full loop over our training data
    for j = 1, training_data:size() do
        updated_x, fs = do_optimisation_step()

        -- NOTE: SGD only estimates the function once, so fs just contains one value
        current_loss = current_loss + fs[1]
    end

    -- Calculate average error for this epoch
    current_loss = current_loss / training_data:size()

    -- Print average loss every tenth epoch
    if i % 10 == 0 then
        print(string.format('After epoch #%3d average loss is %.4f', i, current_loss))
    end

    return current_loss
end

-- Do training
losses = _.map(_.range(1, n_epochs), do_training_epoch)
