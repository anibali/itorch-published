Abalone Age Prediction¶

Aim¶

Given various externally measurable attributes of an abalone, learn how to determine its age.

function visualise_model(model)
    -- Export graph to a temporary location. This produces
    -- a .svg and a .dot file
    local filename = os.tmpname()
    graph.dot(model.fg, 'Title', filename)
    
    -- Delete the .dot file
    os.remove(filename .. '.dot')

    -- Read and base 64 encode the model graph SVG image
    local file = io.open(filename .. '.svg', 'r')
    local image_b64 = base64.encode(file:read('*all'))
    file:close()
    
    -- Delete the .svg file
    os.remove(filename .. '.svg')

    -- Display the graph
    itorch.html('<img src=data:image/svg+xml;base64,' .. image_b64 .. '>')
end

Data set¶

Source

The data set contains 4177 instances of 8 abalone attributes and the corresponding age of the abalone (determined by cutting the shell of the abalone and counting rings).

The table below describes each of the fields.

No. Name Data Type Measurement Unit Description
1 Sex nominal -- M, F, and I (infant)
2 Length continuous mm Longest shell measurement
3 Diameter continuous mm perpendicular to length
4 Height continuous mm with meat in shell
5 Whole weight continuous grams whole abalone
6 Shucked weight continuous grams weight of meat
7 Viscera weight continuous grams gut weight (after bleeding)
8 Shell weight continuous grams after being dried
9 Rings integer -- +1.5 gives the age in years

local file = csv.open("abalone.data")

local raw_data = {}
for record in file:lines() do
    table.insert(raw_data, record)
end

print('Raw data #1 = ', raw_data[1])

local sexes_onehot = {
    M = {1, 0, 0},
    F = {0, 1, 0},
    I = {0, 0, 1}
}

all_inputs = torch.Tensor(_.map(raw_data, function(i, record)
    return _.append(sexes_onehot[record[1]], _.slice(record, 2, 8))
end))

all_outputs = torch.Tensor(_.map(raw_data, function(i, record)
    return record[9]
end))

-- Normalize continuous-valued fields of input
_.invoke(_.range(4, 10), function(i)
    local mean = all_inputs[{{}, i}]:mean()
    local std = all_inputs[{{}, i}]:std()
    all_inputs[{{}, i}]:add(-mean)
    all_inputs[{{}, i}]:div(std)
end)

-- Normalize outputs
local mean = all_outputs:mean()
local std = all_outputs:std()
all_outputs:add(-mean)
all_outputs:div(std)

function output_to_years(output)
    return output * std + mean + 1.5
end

print('Input #1:')
print(all_inputs[1])
print('Output #1:')
print(all_outputs[1])

print()
print(all_inputs:size()[1] .. ' examples loaded successfully')

local n_inputs = 10
local n_outputs = 1
local n_hidden = 10
local n_hidden_layers = 2

local input_layer = nn.Linear(n_inputs, n_hidden)()
local last_hidden_layer = nn.Tanh()(input_layer)

for i = 1, n_hidden_layers - 1 do
    last_hidden_layer = nn.Tanh()(nn.Linear(n_hidden, n_hidden)(last_hidden_layer))
end

local output_layer = nn.Linear(n_hidden, n_outputs)(nn.Dropout(0.5)(last_hidden_layer))

model = nn.gModule({input_layer}, {output_layer})

In [9]:
-- Get the model parameters and their gradients
-- x = flattened model parameters
local x, dloss_dx = model:getParameters()

-- Keep track of current training example index between
-- calls to the evaluation function
local index = 0

-- Define evaluation function as a function that closes
-- over the current sample index, model parameters and
-- their gradients, the model, the criterion, and the
-- training dataset
local feval = function(x_new)
    if x ~= x_new then
        x:copy(x_new)
    end
    
    -- Pull a sample from the training dataset
    index = index + 1
    if index > train_data:size() then index = 1 end
    local input = train_data.inputs[{{index}}]
    local target_output = train_data.outputs[{{index}}]
    
    -- Zero the gradient
    dloss_dx:zero()
    
    -- Forward pass to calculate loss
    local model_output = model:forward(input)
    local loss = criterion:forward(model_output, target_output)
    
    -- Backward pass to modify weights (backpropagation)
    local dloss_dout = criterion:backward(model_output, target_output)
    model:backward(input, dloss_dout)

    return loss, dloss_dx
end

-- Here we will use stochastic gradient descent (SGD)

-- Initialise optimiser state with SGD parameters
local state = {
    learningRate = 1e-2,
    learningRateDecay = 0,
    momentum = 0.2
}

-- Define optimisation step as a function that closes over
-- the evaluation function, model parameters, and optimiser
-- state
function do_optimisation_step()
    return optim.sgd(feval, x, state)
end

-- The number of times to loop over the training dataset
local n_epochs = 400

local function do_training_epoch(i)
    -- This variable is used to estimate the average loss
    local current_loss = 0

    -- Each epoch is a full loop over our training data
    for j = 1, train_data:size() do
        updated_x, fs = do_optimisation_step()

        -- NOTE: SGD only estimates the function once, so fs just contains one value
        current_loss = current_loss + fs[1]
    end

    -- Calculate average error for this epoch
    current_loss = current_loss / train_data:size()

    -- Print average loss every tenth epoch
    if i % 10 == 0 then
        print(string.format('After epoch #%3d average loss is %.6f', i, current_loss))
    end

    return current_loss
end

-- Do training
losses = _.times(n_epochs, do_training_epoch)
